#include <iostream>
using namespace ::std;

double multiply_positive_elements(double A[10],int n)
{
	/*предусловие: дано A[10] - массив из 10 элементов-вещественных чисел и n - целое число элементов массива
	  постусловие: функция возвращает произведение всех положительных элементов массива*/
	double p_positive = 1;
	for (int i = 0; i < n; i++)
		if (A[i] > 0)
			p_positive = p_positive * A[i];
	return p_positive;
}

double multiply_negative_elements(double A[10], int n)
{
	/*предусловие: дано A[10] - массив из 10 элементов-вещественных чисел и n - целое число элементов массива
	  постусловие: функция возвращает произведение всех отрицательных элементов массива*/
	double p_negative = 1;
	for (int i = 0; i < n; i++)
		if (A[i] < 0)
			p_negative = p_negative * A[i];
	return p_negative;
}

void display_array(double A[], int n)
{
	/*предусловие: дан массив А, состоящий из вещественных чисел и целое число n элементов массива
	  постусловие: функция выводит на консоль каждый элемент массива через пробел*/
	for (int i = 0; i < n; i++) cout << A[i] << " ";
}

void generate_array_B(double A[], double B[], int n)
{
	/*предусловие: даны массивы A и B и n - целое число элементов массивов (A и B одинаковы по размеру)
	  постусловие: функция составляет массив B из элементов массива A, умножая все положительные элемен
	for (int i = 0; i < n; i++) //цикл для назначения элементов массива B
	{
		if (A[i] > 0)
			B[i] = A[i] * 2;
		else
			B[i] = A[i];
	};
}

void main()
{
	double A[10], B[10]; //объявление массивов A[10] и B[10]
	system("chcp 1251");

	cout << "Введите 10 вещественных чисел через пробел: ";
	for (int i = 0; i < 10; i++) //цикл для назначения элементов массива A,
		cin >> A[i];			 //запрашивая у пользователя их ввод			  

	cout << "Произведение положительных элементов массива A: " << multiply_positive_elements(A,10) << endl; //вызов функции multiply_positive_elements(A) для нахождения произведения положительных элементов массива A
	cout << "Произведение отрицательных элементов массива A: " << multiply_negative_elements(A,10) << endl; //вызов функции multiply_negative_elements(A) для нахождения произведения отрицательных элементов массива A

	cout << "Массив A: ";
	display_array(A,10); //цикл для вывода элементов массива A на консоль
	cout << endl;
	
	generate_array_B(A, B, 10);

	cout << "Массив B: "; 
	display_array(B,10); //цикл для вывода элементов массива A на консоль
}
